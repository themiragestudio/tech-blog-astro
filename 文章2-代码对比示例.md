**React 版本**
```jsx
// Counter.jsx
import { useState } from 'react';
import './Counter.css';

function Counter() {
  const [count, setCount] = useState(0);
  
  const increment = () => {
    setCount(count + 1);
  };
  
  const reset = () => {
    setCount(0);
  };
  
  return (
    <div className="counter">
      <h2>计数器: {count}</h2>
      <button onClick={increment}>增加</button>
      <button onClick={reset}>重置</button>
    </div>
  );
}

export default Counter;

// ContactForm.jsx
import { useState } from 'react';

function ContactForm() {
  const [formData, setFormData] = useState({
    name: '',
    email: '',
    message: ''
  });
  
  const handleChange = (e) => {
    const { name, value } = e.target;
    setFormData(prev => ({
      ...prev,
      [name]: value
    }));
  };
  
  const handleSubmit = (e) => {
    e.preventDefault();
    console.log('提交数据:', formData);
    // 这里通常是API调用
  };
  
  return (
    <form onSubmit={handleSubmit}>
      <div>
        <label htmlFor="name">姓名:</label>
        <input
          type="text"
          id="name"
          name="name"
          value={formData.name}
          onChange={handleChange}
          required
        />
      </div>
      
      <div>
        <label htmlFor="email">邮箱:</label>
        <input
          type="email"
          id="email"
          name="email"
          value={formData.email}
          onChange={handleChange}
          required
        />
      </div>
      
      <div>
        <label htmlFor="message">消息:</label>
        <textarea
          id="message"
          name="message"
          value={formData.message}
          onChange={handleChange}
          rows="4"
        />
      </div>
      
      <button type="submit">提交</button>
    </form>
  );
}

export default ContactForm;
```

import { useState } from 'react';

function AuthPanel() {
  const [isLoggedIn, setIsLoggedIn] = useState(false);
  const [user, setUser] = useState(null);
  
  const login = () => {
    setIsLoggedIn(true);
    setUser({ name: 'Alice', role: 'admin' });
  };
  
  const logout = () => {
    setIsLoggedIn(false);
    setUser(null);
  };
  
  if (isLoggedIn) {
    return (
      <div className="user-panel">
        <p>欢迎回来, {user.name}!</p>
        <p>角色: {user.role}</p>
        <button onClick={logout}>登出</button>
      </div>
    );
  }
  
  return (
    <div className="login-prompt">
      <p>请先登录</p>
      <button onClick={login}>登录</button>
    </div>
  );
}

export default AuthPanel;

### 2.2 列表渲染

**Svelte 版本**
```svelte
<script>
  let todos = $state([
    { id: 1, text: '学习 Svelte', completed: true },
    { id: 2, text: '学习 React', completed: false },
    { id: 3, text: '写博客文章', completed: false }
  ]);
  
  function toggleTodo(id) {
    todos = todos.map(todo => 
      todo.id === id ? { ...todo, completed: !todo.completed } : todo
    );
  }
</script>

<ul class="todo-list">
  {#each todos as todo (todo.id)}
    <li class:completed={todo.completed}>
      <input 
        type="checkbox" 
        checked={todo.completed}
        on:change={() => toggleTodo(todo.id)}
      />
      <span>{todo.text}</span>
    </li>
  {/each}
</ul>

<style>
  .completed {
    text-decoration: line-through;
    color: #999;
  }
</style>
```

**React 版本**
```jsx
import { useState } from 'react';
import './TodoList.css';

function TodoList() {
  const [todos, setTodos] = useState([
    { id: 1, text: '学习 Svelte', completed: true },
    { id: 2, text: '学习 React', completed: false },
    { id: 3, text: '写博客文章', completed: false }
  ]);
  
  const toggleTodo = (id) => {
    setTodos(todos.map(todo => 
      todo.id === id ? { ...todo, completed: !todo.completed } : todo
    ));
  };
  
  return (
    <ul className="todo-list">
      {todos.map(todo => (
        <li key={todo.id} className={todo.completed ? 'completed' : ''}>
          <input
            type="checkbox"
            checked={todo.completed}
            onChange={() => toggleTodo(todo.id)}
          />
          <span>{todo.text}</span>
        </li>
      ))}
    </ul>
  );
}

export default TodoList;

## 3. 状态管理和副作用

### 3.1 计算属性和副作用

**Svelte 版本**
```svelte
<script>
  let price = $state(100);
  let quantity = $state(2);
  let taxRate = $state(0.1);
  
  // 计算属性
  let subtotal = $derived(price * quantity);
  let tax = $derived(subtotal * taxRate);
  let total = $derived(subtotal + tax);
  
  // 副作用
  $effect(() => {
    console.log('价格变化:', price);
    console.log('数量变化:', quantity);
    console.log('总计:', total);
    
    // 清理函数
    return () => {
      console.log('清理副作用');
    };
  });
</script>

<div class="calculator">
  <div>
    <label>单价: <input type="number" bind:value={price} /></label>
  </div>
  <div>
    <label>数量: <input type="number" bind:value={quantity} /></label>
  </div>
  <div>
    <label>税率: <input type="number" bind:value={taxRate} step="0.01" /></label>
  </div>
  
  <div class="results">
    <p>小计: ${subtotal.toFixed(2)}</p>
    <p>税费: ${tax.toFixed(2)}</p>
    <p>总计: <strong>${total.toFixed(2)}</strong></p>
  </div>
</div>
```

**React 版本**
```jsx
import { useState, useMemo, useEffect } from 'react';

function Calculator() {
  const [price, setPrice] = useState(100);
  const [quantity, setQuantity] = useState(2);
  const [taxRate, setTaxRate] = useState(0.1);
  
  // 计算属性
  const subtotal = useMemo(() => price * quantity, [price, quantity]);
  const tax = useMemo(() => subtotal * taxRate, [subtotal, taxRate]);
  const total = useMemo(() => subtotal + tax, [subtotal, tax]);
  
  // 副作用
  useEffect(() => {
    console.log('价格变化:', price);
    console.log('数量变化:', quantity);
    console.log('总计:', total);
    
    // 清理函数
    return () => {
      console.log('清理副作用');
    };
  }, [price, quantity, total]);
  
  return (
    <div className="calculator">
      <div>
        <label>
          单价: 
          <input 
            type="number" 
            value={price}
            onChange={(e) => setPrice(Number(e.target.value))}
          />
        </label>
      </div>
      <div>
        <label>
          数量: 
          <input 
            type="number" 
            value={quantity}
            onChange={(e) => setQuantity(Number(e.target.value))}
          />
        </label>
      </div>
      <div>
        <label>
          税率: 
          <input 
            type="number" 
            value={taxRate}
            step="0.01"
            onChange={(e) => setTaxRate(Number(e.target.value))}
          />
        </label>
      </div>
      
      <div className="results">
        <p>小计: ${subtotal.toFixed(2)}</p>
        <p>税费: ${tax.toFixed(2)}</p>
        <p>总计: <strong>${total.toFixed(2)}</strong></p>
      </div>
    </div>
  );
}

export default Calculator;
```

## 4. 组件通信

### 4.1 父子组件通信

**Svelte 版本**
```svelte
<!-- Parent.svelte -->
<script>
  import Child from './Child.svelte';
  
  let parentMessage = $state('来自父组件的消息');
  let childMessage = $state('');
  
  function handleChildEvent(message) {
    childMessage = message;
  }
</script>

<div class="parent">
  <h2>父组件</h2>
  <p>子组件消息: {childMessage}</p>
  
  <Child 
    message={parentMessage}
    on:customEvent={handleChildEvent}
  />
</div>

<!-- Child.svelte -->
<script>
  import { createEventDispatcher } from 'svelte';
  
  export let message;
  const dispatch = createEventDispatcher();
  
  function sendMessage() {
    dispatch('customEvent', '来自子组件的消息');
  }
</script>

<div class="child">
  <h3>子组件</h3>
  <p>父组件消息: {message}</p>
  <button on:click={sendMessage}>发送消息给父组件</button>
</div>
```

**React 版本**
```jsx
// Parent.jsx
import { useState } from 'react';
import Child from './Child';

function Parent() {
  const [parentMessage] = useState('来自父组件的消息');
  const [childMessage, setChildMessage] = useState('');
  
  const handleChildEvent = (message) => {
    setChildMessage(message);
  };
  
  return (
    <div className="parent">
      <h2>父组件</h2>
      <p>子组件消息: {childMessage}</p>
      
      <Child 
        message={parentMessage}
        onCustomEvent={handleChildEvent}
      />
    </div>
  );
}

export default Parent;

// Child.jsx
function Child({ message, onCustomEvent }) {
  const sendMessage = () => {
    onCustomEvent('来自子组件的消息');
  };
  
  return (
    <div className="child">
      <h3>子组件</h3>
      <p>父组件消息: {message}</p>
      <button onClick={sendMessage}>发送消息给父组件</button>
    </div>
  );
}

export default Child;
```

## 5. 迁移辅助脚本

### 5.1 基础转换脚本

```python
# svelte_to_react.py
import re
import os

class SvelteToReactConverter:
    def __init__(self):
        self.conversion_patterns = [
            # 状态声明
            (r'let\s+(\w+)\s*=\s*\$state\(', r'const [\1, set\1] = useState('),
            
            # 计算属性
            (r'let\s+(\w+)\s*=\s*\$derived\(', r'const \1 = useMemo(() => '),
            
            # 副作用
            (r'\$effect\(', r'useEffect(() => {'),
            
            # 事件处理
            (r'on:click=', r'onClick='),
            (r'on:input=', r'onChange='),
            (r'on:submit=', r'onSubmit='),
            (r'on:change=', r'onChange='),
            
            # 类名绑定
            (r'class=', r'className='),
            (r'class:(\w+)=', r'className={\1 ? "\1" : ""}'),
            
            # 条件渲染
            (r'\{#if\s+(.+?)\}', r'{\1 && ('),
            (r'\{:else\}', r') : ('),
            (r'\{/if\}', r')}'),
            
            # 列表渲染
            (r'\{#each\s+(.+?)\s+as\s+(.+?)\}', r'{\1.map(\2 => ('),
            (r'\{/each\}', r'))}'),
            
            # 绑定
            (r'bind:value=', r'value='),
            (r'bind:checked=', r'checked='),
        ]
        
        self.import_statements = {
            'useState': 'import { useState } from \'react\';',
            'useEffect': 'import { useEffect } from \'react\';',
            'useMemo': 'import { useMemo } from \'react\';',
        }
    
    def convert_file(self, svelte_path, react_path):
        with open(svelte_path, 'r', encoding='utf-8') as f:
            content = f.read()
        
        # 应用转换规则
        converted_content = content
        for pattern, replacement in self.conversion_patterns:
            converted_content = re.sub(pattern, replacement, converted_content)
        
        # 提取需要导入的hooks
        imports_needed = []
        for hook, import_stmt in self.import_statements.items():
            if hook in converted_content:
                imports_needed.append(import_stmt)
        
        # 构建React组件
        component_name = os.path.basename(svelte_path).replace('.svelte', '')
        react_code = f"""import React from 'react';
{chr(10).join(imports_needed)}

function {component_name}() {{
  {converted_content}
}}

export default {component_name};
"""
        
        # 写入文件
        with open(react_path, 'w', encoding='utf-8') as f:
            f.write(react_code)
        
        print(f"✅ 已转换: {svelte_path} -> {react_path}")
        return True
